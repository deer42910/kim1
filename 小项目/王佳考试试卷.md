```
public class interationSort{
  public static void sort(int[] arr){
      int len = arr.length;
      if(len<=1){
          return;
      }
      for(int i = 1;i<len;i++){
         int value = arr[i];
         int j = i-1;
         for(;j>=0;j--){
             if(value<arr[j])
                arr[j+1]=arr[j]
             else
                break;
         }
         arr[j+1]=value;
         //输出每一次排序后的结果
         System.out.println(Arrays.toString(arr));
      }
  }
}
```

```java
public class Rabbit2 {
    public static int rabbitCount(int months){
        if (months==1||months==2){
            return 1;
            //第一二月都只有一对兔子
        }else {
            // 斐波那契数列公式
            return rabbitCount(months-1)+rabbitCount(months-2);
        }
    }
    public static void main(String[] args) {
        int count = rabbitCount(10);
        System.out.println("10个月共有"+count+"个兔子");
    }
}
```
